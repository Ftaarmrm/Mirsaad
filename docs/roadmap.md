### تفاصيل تقنية

```typescript
// نظام Webhooks
interface WebhookConfig {
  id: string;
  userId: string;
  url: string;
  events: WebhookEvent[];
  secret: string;
  isActive: boolean;
}

type WebhookEvent = 
  | 'article.created'
  | 'article.analyzed'
  | 'alert.triggered'
  | 'event.created';

// إرسال Webhook
async function sendWebhook(config: WebhookConfig, payload: unknown) {
  const signature = createHmac('sha256', config.secret)
    .update(JSON.stringify(payload))
    .digest('hex');
  
  await fetch(config.url, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-Mirsad-Signature': signature,
    },
    body: JSON.stringify({
      event: payload,
      timestamp: new Date().toISOString(),
    }),
  });
}
```
